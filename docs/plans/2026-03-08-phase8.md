```haskell
{-# LANGUAGE MagicHash #-}
{-# LANGUAGE UnboxedTuples #-}
module Grasp.Continuations
  ( newPromptTag
  , prompt
  , control0
  , pushHandler
  , popHandler
  , peekHandler
  , initHandlerStack
  , handlerStack
  ) where

import GHC.Exts (Any, PromptTag#, newPromptTag#, prompt#, control0#)
import GHC.IO (IO(IO))
import Unsafe.Coerce (unsafeCoerce)
import Data.IORef
import System.IO.Unsafe (unsafePerformIO)

-- | Box a PromptTag# as Any for storage
boxPromptTag :: PromptTag# a -> Any
boxPromptTag t = unsafeCoerce t

-- | Unbox Any back to PromptTag#
unboxPromptTag :: Any -> PromptTag# a
unboxPromptTag t = unsafeCoerce t

-- | Create a new prompt tag (IO wrapper around newPromptTag#)
newPromptTag :: IO Any
newPromptTag = IO $ \s ->
  case newPromptTag# s of
    (# s', tag #) -> (# s', boxPromptTag tag #)

-- | Delimit a computation with a prompt tag
prompt :: Any -> IO Any -> IO Any
prompt tagBox body = IO $ \s ->
  prompt# (unboxPromptTag tagBox) (unIO body) s
  where
    unIO (IO f) = f

-- | Capture the continuation up to the nearest prompt with the given tag
-- The callback receives a function (Any -> IO Any) that resumes the continuation.
control0 :: Any -> ((Any -> IO Any) -> IO Any) -> IO Any
control0 tagBox callback = IO $ \s ->
  control0# (unboxPromptTag tagBox)
    (\k s' -> unIO (callback (\v -> IO (\s'' -> k (unIO (pure v)) s''))) s')
    s
  where
    unIO (IO f) = f

-- | Global handler stack: list of (promptTag, handlerFn) pairs as Any
{-# NOINLINE handlerStack #-}
handlerStack :: IORef [(Any, Any)]
handlerStack = unsafePerformIO (newIORef [])

initHandlerStack :: IO ()
initHandlerStack = writeIORef handlerStack []

pushHandler :: Any -> Any -> IO ()
pushHandler tag handler = modifyIORef' handlerStack ((tag, handler) :)

popHandler :: IO ()
popHandler = modifyIORef' handlerStack tail

peekHandler :: IO (Maybe (Any, Any))
peekHandler = do
  stack <- readIORef handlerStack
  pure $ case stack of
    []    -> Nothing
    (h:_) -> Just h
```

```haskell
-- with-handler: establish a condition handler
eval env (EList [ESym "with-handler", handlerExpr, body]) = do
  handler <- eval env handlerExpr
  tag <- newPromptTag
  pushHandler tag handler
  result <- (do
    r <- prompt tag (eval env body)
    popHandler
    pure r)
    `catch` \(e :: SomeException) -> do
      popHandler
      -- Call handler with (error-string, non-resumable-restart)
      let errStr = mkStr (T.pack (displayException e))
      let noRestart = mkPrim "<no-restart>" (\_ -> error "cannot restart from a Haskell exception")
      apply handler [errStr, noRestart]
  pure result

-- signal: raise a condition to the nearest handler
eval env (EList [ESym "signal", valExpr]) = do
  val <- eval env valExpr
  mh <- peekHandler
  case mh of
    Nothing -> error "signal: no handler installed"
    Just (tag, handler) -> do
      control0 tag $ \k -> do
        popHandler
        let restart = mkPrim "<restart>" (\case
              [v] -> do
                pushHandler tag handler
                r <- k v
                popHandler
                pure r
              _ -> error "restart expects one argument")
        apply handler [val, restart]
```

Helper function (add after `graspListToList` or similar helper area):

```haskell
-- | Try to match a runtime value against a pattern expression.
-- Returns Just bindings on match, Nothing on no match.
matchPattern :: Any -> LispExpr -> IO (Maybe (Map.Map T.Text Any))
matchPattern val (ESym "_") = pure (Just Map.empty)  -- wildcard
matchPattern val (ESym name) = pure (Just (Map.singleton name val))  -- variable bind
matchPattern val (EInt n) = do
  v <- forceIfLazy val
  if graspTypeOf v == GTInt && toInt v == fromInteger n
    then pure (Just Map.empty)
    else pure Nothing
matchPattern val (EStr s) = do
  v <- forceIfLazy val
  if graspTypeOf v == GTStr && toStr v == s
    then pure (Just Map.empty)
    else pure Nothing
matchPattern val (EBool b) = do
  v <- forceIfLazy val
  let expected = if b then GTBoolTrue else GTBoolFalse
  if graspTypeOf v == expected
    then pure (Just Map.empty)
    else pure Nothing
matchPattern val (EList []) = do  -- nil pattern
  v <- forceIfLazy val
  if isNil v then pure (Just Map.empty) else pure Nothing
matchPattern val (EList [ESym "cons", hPat, tPat]) = do  -- cons destructuring
  v <- forceIfLazy val
  if isCons v then do
    mh <- matchPattern (toCar v) hPat
    case mh of
      Nothing -> pure Nothing
      Just hBinds -> do
        mt <- matchPattern (toCdr v) tPat
        case mt of
          Nothing -> pure Nothing
          Just tBinds -> pure (Just (Map.union hBinds tBinds))
  else pure Nothing
matchPattern _ _ = pure Nothing
```

Eval clause (add before the generic application clause):

```haskell
-- match: structural pattern matching
eval env (EList (ESym "match" : scrutinee : clauses)) = do
  val <- eval env scrutinee
  let tryMatch [] = error "match: no matching clause"
      tryMatch (EList [pat, body] : rest) = do
        result <- matchPattern val pat
        case result of
          Nothing -> tryMatch rest
          Just bindings -> do
            parentEd <- readIORef env
            childEnv <- newIORef $ parentEd { envBindings = Map.union bindings (envBindings parentEd) }
            eval childEnv body
      tryMatch _ = error "match: each clause must be (pattern body)"
  tryMatch clauses
```

```haskell
{-# LANGUAGE OverloadedStrings #-}
{-# LANGUAGE ScopedTypeVariables #-}
module Main where

import System.Environment (getArgs)
import System.Exit (exitFailure)
import qualified Data.Text as T
import qualified Data.Text.IO as TIO
import Control.Exception (catch, SomeException, displayException)
import Data.IORef (readIORef)
import qualified Data.Map.Strict as Map

import qualified System.Console.Isocline as IC

import Grasp.Types (Env)
import Grasp.Parser
import Grasp.Eval
import Grasp.Printer
import Grasp.HaskellInterop (defaultEnvWithInterop)

main :: IO ()
main = do
  args <- getArgs
  case args of
    [] -> do
      putStrLn "grasp v0.1 — a Lisp on GHC's runtime"
      putStrLn "Type (quit) to exit."
      IC.setHistory ".grasp_history" 200
      IC.setPromptMarker "λ> " ".. "
      env <- defaultEnvWithInterop
      repl env
    [file] -> runFile file
    _ -> do
      putStrLn "Usage: grasp [file.gsp]"
      exitFailure

runFile :: FilePath -> IO ()
runFile file = do
  content <- TIO.readFile file
  case parseFile content of
    Left err -> do
      putStrLn $ "parse error: " <> show err
      exitFailure
    Right exprs -> do
      env <- defaultEnvWithInterop
      results <- mapM (\expr ->
        (Right <$> eval env expr)
          `catch` \(e :: SomeException) -> do
            putStrLn $ "error: " <> displayException e
            pure (Left ())) exprs
      case [v | Right v <- results] of
        [] -> pure ()
        vs -> putStrLn (printVal (last vs))

repl :: Env -> IO ()
repl env = do
  input <- IC.readlineEx "λ> " (Just (completer env)) Nothing
  if null input
    then repl env  -- empty line, continue
    else if T.strip (T.pack input) == "(quit)"
      then putStrLn "Bye."
      else do
        -- Check for balanced parens; if unbalanced, read more lines
        let full = collectMultiline input
        case parseLisp (T.pack full) of
          Left err -> putStrLn $ "parse error: " <> show err
          Right expr -> do
            result <- (Right <$> eval env expr)
              `catch` \(e :: SomeException) -> pure (Left (displayException e))
            case result of
              Right val -> putStrLn (printVal val)
              Left err  -> putStrLn $ "error: " <> err
        repl env

-- | Simple paren-balance check for multi-line input
collectMultiline :: String -> String
collectMultiline s
  | parenBalance s >= 0 && parenBalance s == 0 = s
  | otherwise = s  -- isocline handles continuation via setPromptMarker

parenBalance :: String -> Int
parenBalance = foldl (\acc c -> case c of '(' -> acc+1; ')' -> acc-1; _ -> acc) 0

-- | Tab completer: completes env bindings
completer :: Env -> IC.CompletionEnv -> String -> IO ()
completer env cenv prefix = do
  ed <- readIORef env
  let names = Map.keys (envBindings ed)
  let matches = filter (T.isPrefixOf (T.pack prefix)) names
  IC.completeWord cenv prefix Nothing $ \_ ->
    map (\n -> IC.Completion (T.unpack n) (T.unpack n) "") matches
```

Note: The exact isocline API may vary. The implementer should check the isocline Haskell docs and adjust. The key points are:
- Use `IC.readline` or `IC.readlineEx` for input

typeOfOp :: [Any] -> IO Any
typeOfOp [v] = do

  pure $ mkStr (T.pack (showGraspType (graspTypeOf v')))
typeOfOp _ = error "type-of expects one argument"

inspectOp :: [Any] -> IO Any
inspectOp [v] = do
  v' <- forceIfLazy v
  let typeName = mkStr (T.pack (showGraspType (graspTypeOf v')))
  case unpackClosure# v' of
    (# _, ptrs, nptrs #) ->
      let np = mkInt (sizeofArray ptrs)
          nnp = mkInt (sizeofByteArray nptrs)
      in pure $ foldr mkCons mkNil
          [ mkSym "type", typeName
          , mkSym "pointers", np
          , mkSym "non-pointers", nnp
          ]
  where

    sizeofArray _ = 0  -- simplified; use GHC.Exts.sizeofArray# for real count

    sizeofByteArray _ = 0
inspectOp _ = error "inspect expects one argument"

gcStatsOp :: [Any] -> IO Any
gcStatsOp [] = do
  enabled <- Stats.getRTSStatsEnabled
